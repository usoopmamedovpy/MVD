async function confirmCancel() { const contact = localStorage.getItem('mvd_contact'); if (!contact || !activeBooking) return; // Сохраняем данные для повторного заполнения const prevName = activeBooking.name; const prevSurname = activeBooking.surname; try { const res = await fetch(`${API_URL}/api/cancel`, { ... }); const data = await res.json(); if (data.success) { localStorage.removeItem('mvd_contact'); activeBooking = null; closeCancelModal(); // Предзаполняем форму document.getElementById('name').value = prevName; document.getElementById('surname').value = prevSurname; document.getElementById('contact').value = contact; showFormScreen(); await loadDates(); } } }
